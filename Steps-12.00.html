Adding DAX Measures 12

First we will create a  Measure Table which will contain all the measures we are using for this dashboard.

  • For Creating a Measure Table,Click ‘Enter Data’ in the Home Ribbon,the window will open up.Rename the Table as ‘Measure Table’ and Measure Table will be created.

  • Then Right Click on the Measure Table and Select New measure.

  • In the window that opened up, copy and paste the DAX  formula below to create the first measure.

    .Signups per wk kpi =
    VAR sumAttendees =
       AVERAGEX ( 'EventBrite_Events', COUNT ( EventBrite_EventAttendees[UserId] ) )
    VAR weeks =
       DISTINCTCOUNT ( 'Date Table'[Week Number] )
    RETURN
       DIVIDE ( sumAttendees, weeks )


    Then format the measure, Click on the measure and under Measure tools -> Format

    Similarly add all the following DAX measures in order. You can also use the PBIX file attached which has all the DAX measures.

    1. .No Of Events =  CALCULATE ( DISTINCTCOUNT ( EventBrite_Events[EventId] ) )

    2. Events Per Wk = AVERAGEX (   KEEPFILTERS ( VALUES ( 'Date Table'[For the Week] ) ),   CALCULATE ( [.No Of Events] ))

    1. .Number Of Signups =   CALCULATE ( COUNT ( EventBrite_EventAttendees[UserId] ) )

    2. .Total Signups PW =   CALCULATE ( [.Number Of Signups], DATEADD ( 'Date Table'[Date], -7, DAY ) )

    3. signups WOW =IF (   ISBLANK ( [.Total Signups PW] ),   0,   DIVIDE ( [.Number Of Signups] - [.Total Signups PW], [.Total Signups PW] ))

    1. SignupsPerEvent = 

    VAR signups =    COUNT ( EventBrite_EventAttendees[UserId] ) 

    VAR tot_event =

       CALCULATE (

           DISTINCTCOUNT ( EventBrite_EventAttendees[EventId] ),

           FILTER ( EventBrite_EventAttendees, RELATED ( EventBrite_Events[EventId] ) )

       )

    RETURN

       DIVIDE ( signups, tot_event, 0 )

    1. .Events Graded Per Week =

    CALCULATE (

       DISTINCTCOUNT ( EventBrite_Events[EventId] ),

       FILTER ( EventBrite_Events, EventBrite_Events[Topics] <> BLANK () )

    )

    1. Graded Per Week = AVERAGEX(

        KEEPFILTERS(VALUES('Date Table'[For the Week])),

        CALCULATE([.Events Graded Per Week]))

    1. DaysToGrade=

    AVERAGEX (

        EventBrite_Events,

        DATEDIFF (

            EventBrite_Events[StartDate],

            EventBrite_Events[NewEventAddedOn],

            DAY    ))

    1. DaysToGradeCurrWeek = CALCULATE([DaysForVideo])

    2. DaysToGradePrevWeek = CALCULATE ( [DaysForVideo], DATEADD ( 'Date Table'[Date], -7, DAY))

    3. DaysToGradeWoW = DIVIDE (   [DaysToGradeCurrWeek] - [DaysToGradePrevWeek],

       [DaysToGradePrevWeek],    0)

    1. Rolling 4 weeks events =

    VAR past4weeksAvg =

       CALCULATE (

           COUNT ( EventBrite_Events[EventId] ),

           DATESINPERIOD ( 'Date Table'[Date], LASTDATE ( 'Date Table'[Date] ), -28, DAY )

       ) / 4

    VAR rolling4wkPercent =

       DIVIDE ( ( [.No Of Events] - past4weeksAvg ), past4weeksAvg, 0 )

    RETURN

       rolling4wkPercent

    1. Rolling 8 weeks events =

    VAR past8weeksAvg =

        CALCULATE (

            COUNT ( EventBrite_Events[EventId] ),

            DATESINPERIOD ( 'Date Table'[Date], LASTDATE ( 'Date Table'[Date] ), -56, DAY )

        ) / 8

    VAR rolling8wkPercent =

        DIVIDE ( ( [.No Of Events per .Week] - past8weeksAvg ), past8weeksAvg, 0 )

    RETURN

        rolling8wkPercent

    1. signup Rolling 4 weeks =

    VAR past4weeksAvg =

       CALCULATE (

           COUNT ( EventBrite_EventAttendees[UserId] ),

           DATESINPERIOD ( 'Date Table'[Date], LASTDATE ( 'Date Table'[Date] ), -28, DAY )) / 4

    VAR rolling4wkPercent =

       DIVIDE (

           ( COUNT ( EventBrite_EventAttendees[UserId] ) - past4weeksAvg ),

           past4weeksAvg,  0  )

    RETURN

       rolling4wkPercent

    1. signups Rolling 8 weeks =

    VAR past8weeksAvg =

       CALCULATE (

           COUNT ( EventBrite_EventAttendees[UserId] ),

           DATESINPERIOD ( 'Date Table'[Date], LASTDATE ( 'Date Table'[Date] ), -56, DAY )

       ) / 8

    VAR rolling8wkPercent =

       DIVIDE ( ( [.Number Of Signups] - past8weeksAvg ), past8weeksAvg, 0 )

    RETURN

       rolling8wkPercent

    1. No of weeks =

    VAR start_date =

        MIN ( EventBrite_EventAttendees[CreatedDate] )

    VAR end_date =

        MAX ( EventBrite_EventAttendees[CreatedDate] )

    RETURN

        DATEDIFF ( start_date, end_date, WEEK )

     

    1. Total signups per week =

    DIVIDE ( COUNT ( EventBrite_EventAttendees[UserId] ), [No of weeks], 0 )

    1. .Number Of Signups Per Event =

    CALCULATE (

       COUNT ( EventBrite_EventAttendees[UserId] ),

       FILTER ( EventBrite_Events, EventBrite_Events[Name] <> "NULL" )

    )

    1. Rolling 4 signups event =

    VAR sum_in_period =

       CALCULATE (

           [.Number Of Signups Per Event],

           DATESINPERIOD (

               EventBrite_Events[StartDate],

               LASTDATE ( EventBrite_Events[StartDate] ), -28, DAY

           )

       )

    VAR weeks_in_period = 4

    VAR past4weeks =

       DIVIDE ( sum_in_period, weeks_in_period, 0 )

    RETURN

       DIVIDE (

           past4weeks - [Total signups per week],[Total signups per week], 0

       )

    1. Rolling 8 signups event =

    VAR sum_in_period =

       CALCULATE (

           [.Number Of Signups Per Event],

           DATESINPERIOD (

               EventBrite_Events[StartDate],

               LASTDATE ( EventBrite_Events[StartDate] ), -56, DAY

           ),

           EventBrite_Events[StartDate] < TODAY ()

       )

    VAR weeks_in_period = 8

    VAR past8weeks =

       DIVIDE ( sum_in_period, weeks_in_period, 0 )

    RETURN

       DIVIDE (

           past8weeks - [Total signups per week],

           [Total signups per week],

           0

       )

    1. .Arrow Indicator WoW = SWITCH(TRUE(),[.WoW change in signups]< 0,UNICHAR(9660),UNICHAR(9650))

    3. Color WoW Grade = IF([.Arrow Indicator grade WoW] = UNICHAR(9660), "Red","Green")
